import csv

import pandas as pd

# Load the CSV file into a DataFrame
df = pd.read_csv('DataLaboratory.csv')

# Group the DataFrame by the 'groupid' column and compute the mean of the 'value' column for each group
grouped_by_elo = df.groupby('modularity_class')['elo'].mean()
grouped_by_betweeness = df.groupby('modularity_class')['betweenesscentrality'].mean()
grouped_by_closness = df.groupby('modularity_class')['closnesscentrality'].mean()
grouped_by_Degree = df.groupby('modularity_class')['Degree'].mean()
grouped_by_count = df.modularity_class.value_counts()
print(grouped_by_elo)
print(grouped_by_betweeness)
print(grouped_by_closness)
print(grouped_by_Degree)
print(grouped_by_count)
